## Component Details

### TripsReport.php
**Properties:**
- `$month` (int): Selected month (default: current month)
- `$year` (int): Selected year (default: current year)
- `$trips` (collection): Filtered trips data
- `$summary` (array): Calculated summary metrics

**Methods:**
- `mount()`: Set default month/year
- `updatedMonth()`/`updatedYear()`: Refresh data when filters change
- `fetchData()`: Main data fetching method
- `calculateSummary()`: Compute all required metrics
- `printReport()`: Trigger window.print event
- `getMonths()`: Helper for month dropdown
- `getYears()`: Helper for year dropdown

**Query Optimization:**
- Eager load party, truck, driver relationships
- Use whereMonth() and whereYear() for filtering
- Calculate aggregates in database where possible
- Avoid N+1 with proper relationship loading

**Implementation Details:**
```php
use App\Models\Trip;
use App\Models\Party;
use App\Models\Truck;
use App\Models\Driver;
use App\Models\TripExpense;
use Illuminate\Support\Facades\DB;
use Carbon\Carbon;
use Livewire\Component;

class TripsReport extends Component
{
    public int $month;
    public int $year;
    public $trips;
    public $summary = [];

    public function mount()
    {
        $this->month = Carbon::now()->month;
        $this->year = Carbon::now()->year;
        $this->fetchData();
    }

    public function updatedMonth()
    {
        $this->fetchData();
    }

    public function updatedYear()
    {
        $this->fetchData();
    }

    public function fetchData()
    {
        $this->trips = Trip::with(['party', 'truck', 'driver'])
            ->whereMonth('start_date', $this->month)
            ->whereYear('start_date', $this->year)
            ->orderBy('start_date', 'desc')
            ->get();

        $this->calculateSummary();
    }

    public function calculateSummary()
    {
        $trips = $this->trips;

        // Basic counts
        $totalTrips = $trips->count();
        $completedTrips = $trips->whereIn('status', ['completed', 'pod_received', 'pod_submitted', 'settled'])->count();
        $ongoingTrips = $trips->whereIn('status', ['pending', 'start'])->count();
        $cancelledTrips = $trips->where('status', 'cancelled')->count(); // if exists

        // Financial calculations - using eager loaded relationships to avoid N+1
        $totalPartyAmount = $trips->sum('freight_amount');
        
        // Get expenses for all trips in one query to avoid N+1
        $tripIds = $trips->pluck('id')->toArray();
        $totalExpenses = 0;
        $expenseBreakdown = [];
        
        if (!empty($tripIds)) {
            $expenses = TripExpense::whereIn('trip_id', $tripIds)
                ->where('expense_category', 'trip')
                ->get();
            
            $totalExpenses = $expenses->sum('amount');
            
            // Group by expense type
            $expenseBreakdown = $expenses->groupBy('expense_type')
                ->mapWithKeys(function ($group) {
                    return [$group->first()->expense_type => $group->sum('amount')];
                })
                ->toArray();
        }

        $profitLoss = $totalPartyAmount - $totalExpenses;

        // Average trip profit
        $averageTripProfit = $totalTrips > 0 ? $profitLoss / $totalTrips : 0;

        // Top performing route
        $topRoute = null;
        if ($totalTrips > 0) {
            $topRoute = $trips->whereNotNull('origin')
                ->whereNotNull('destination')
                ->groupBy('origin', 'destination')
                ->select(DB::raw('COUNT(*) as trip_count, origin, destination'))
                ->orderByDesc('trip_count')
                ->first()
                ?->toArray();
        }

        // Monthly overview table data
        $monthlyOverview = $trips->map(function ($trip) {
            $tripExpenses = $trip->expenses->where('expense_category', 'trip')->sum('amount');
            return [
                'id' => $trip->id,
                'date' => $trip->start_date->format('d M Y'),
                'party' => $trip->party?->name ?? $trip->party_name,
                'truck' => $trip->truck?->truck_number ?? $trip->truck_name,
                'driver' => $trip->driver?->name ?? $trip->driver_name,
                'origin' => $trip->origin,
                'destination' => $trip->destination,
                'freight_amount' => $trip->freight_amount,
                'expenses' => $tripExpenses,
                'profit' => $trip->freight_amount - $tripExpenses,
                'status' => $trip->status,
            ];
        });

        $this->summary = [
            'total_trips' => $totalTrips,
            'completed_trips' => $completedTrips,
            'ongoing_trips' => $ongoingTrips,
            'cancelled_trips' => $cancelledTrips,
            'total_party_amount' => $totalPartyAmount,
            'total_expenses' => $totalExpenses,
            'profit_loss' => $profitLoss,
            'expense_breakdown' => $expenseBreakdown,
            'average_trip_profit' => round($averageTripProfit, 2),
            'top_route' => $topRoute,
            'monthly_overview' => $monthlyOverview,
        ];
    }

    public function printReport()
    {
        $this->dispatch('printReport');
    }

    public function getMonths()
    {
        return [
            1 => 'January', 2 => 'February', 3 => 'March', 4 => 'April',
            5 => 'May', 6 => 'June', 7 => 'July', 8 => 'August',
            9 => 'September', 10 => 'October', 11 => 'November', 12 => 'December'
        ];
    }

    public function getYears()
    {
        $currentYear = Carbon::now()->year;
        return range($currentYear - 5, $currentYear + 5);
    }

    public function render()
    {
        return view('livewire.admin.reports.trips-report');
    }
}
```
